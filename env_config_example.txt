# 阿里云OSS配置环境变量示例
# 复制以下环境变量到您的系统环境或.env文件中

# OSS认证模式配置 (推荐在ECS上使用RAM Role)
export OSS_USE_ECS_RAM_ROLE="true"  # true: 使用ECS RAM Role, false: 使用AKSK

# OSS AccessKey配置 (仅当OSS_USE_ECS_RAM_ROLE=false时需要)
export OSS_ACCESS_KEY_ID="your_access_key_id_here"
export OSS_ACCESS_KEY_SECRET="your_access_key_secret_here"

# OSS服务配置
export OSS_ENDPOINT="oss-cn-beijing.aliyuncs.com"
export OSS_BUCKET_NAME="your-bucket-name"

# API认证配置
export API_USERNAME="emoji_user"
export API_PASSWORD="your_secure_password_here"

# 可选配置
export OSS_CUSTOM_DOMAIN=""
export OSS_USE_HTTPS="true"
export OSS_EMOJI_ROOT_PATH="assets/"

# Windows用户请使用set命令：
# set OSS_ACCESS_KEY_ID=your_access_key_id_here
# set OSS_ACCESS_KEY_SECRET=your_access_key_secret_here

# 使用方法：
# 1. Linux/Mac: source env_config_example.txt (修改后)
# 2. Windows: 直接运行修改后的set命令
# 3. 或者使用python-dotenv库读取.env文件
